# Extracting topics with their word distributions from the LDA model for text
lda_text_topics = ldamodel_text.print_topics(num_topics=19, num_words=15)
text_topic_keywords_list = []

for text_topic_number, text_word_distribution in lda_text_topics:
    # Extracting individual words from the word distribution string for text
    text_words = text_word_distribution.split('+')
    text_words_cleaned = [word.split('*')[1].strip().replace('"', '') for word in text_words]
    text_words_joined = ", ".join(text_words_cleaned)
    text_topic_keywords_list.append((text_topic_number, text_words_joined))

# Creating a DataFrame to display text topics and their keywords
df_text_topic_keywords = pd.DataFrame(text_topic_keywords_list, columns=['Text_Topic', 'Text_Keywords'])
